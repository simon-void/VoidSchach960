WINDOWS
-------

jpackage --type exe \
         --name VoidChess \
         --description "a chess program" \
         --vendor "StephanS" \
         --app-version 3.2.0 \
         --icon about/shortcut-icon.ico \
         --input build/libs --dest build/installer \
         --main-jar voidchess-3.2.0-all.jar \
         --add-modules java.desktop \
         --win-shortcut --win-menu
